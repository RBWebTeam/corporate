// A complete working C program to demonstrate all insertion methods
// on Linked List
#include <stdio.h>
#include <stdlib.h>

// A linked list node
struct Node
{
int data;
struct Node *next;
};

struct Node* head;

void insert(int data){
    
    
   struct Node* temp = (struct Node*) malloc(sizeof(struct Node));
   
   temp->data=data;
   temp->next=head;
   head=temp;
   
}

void print(){
    
    struct Node* temp=head;
    
    while (temp!=NULL){
        printf("%d",temp->data);
        temp=temp->next;
        printf("\n");
    }
}

int main(){
    
    head=NULL;
    insert(2);
    insert(3);
    insert(4);
    print();
}
